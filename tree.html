# Tree

用清晰的层级结构展示信息，可展开或折叠，还集成了树搜索功能。

# 基础用法

通过datatType控制传入数据类型。

通常后端返回的都是平行结构的数据，请将dataType设置为parallel后再传入。

也可以直接传入树结构数据，则无需设置dataType，dataType默认为tree。

:::demo

<template>
  <div>
    <ds-title>传入parallel型数据</ds-title>
    <ds-tree :data="parallelData" dataType='parallel'></ds-tree>
    <ds-title style="marginTop: 16px;">传入tree型数据</ds-title>
    <ds-tree :data="data"></ds-tree>
  </div>
</template>
<script>
  export default {
    data() {
      return {
        parallelData: [
          { id: 1, parentId: 0, label: '一级1' },
          { id: 11, parentId: 1, label: '二级1-1' },
          { id: 111, parentId: 11, label: '二级1-1-1' },
          { id: 112, parentId: 11, label: '二级1-1-2' },
          { id: 2, parentId: 0, label: '一级2' },
          { id: 21, parentId: 2, label: '二级2-1' },
          { id: 22, parentId: 2, label: '二级2-2' },
        ],
        data: [
          {
            id: 1,
            parentId: 0,
            label: '一级1',
            children: [
              {
                id: 11,
                parentId: 1,
                label: '二级1-1',
                children: [
                  { id: 111, parentId: 11, label: '三级1-1-1' },
                  { id: 112, parentId: 11, label: '三级1-1-2'}
                ]
              }
            ]
          },
          {
            id: 2,
            parentId: 0,
            label: '一级2',
            children: [
              { id: 21, parentId: 2, label: '二级2-1'},
              { id: 22, parentId: 2, label: '二级2-2'}
            ]
          }
        ],
      }
    }
  }
</script>

:::

# 懒加载

设置lazy则启用懒加载功能。

同时需要配置懒加载函数load。

又因为异步，load必须return一个Promise，并且把返回数据resolve给组件内部使用。

在load函数中会抛出当前的节点数据node。

通常从node中拿到当前节点的id作为参数去请求懒加载的数据。

:::demo

<template>
  <ds-tree lazy :load="load"></ds-tree>
</template>
<script>
  export default {
    methods: {
      load(node) {
        // 展开节点时触发
        return new Promise((resolve, reject) => {
          // e.g：req('getData', { id: node.data.id }).then(data => resolve(data))
          setTimeout(() => {
            resolve([
              { id: 1, parentId: 0, label: '一级1' },
              { id: 11, parentId: 1, label: '二级1-1' },
              { id: 111, parentId: 11, label: '二级1-1-1' },
              { id: 112, parentId: 11, label: '二级1-1-2' },
              { id: 2, parentId: 0, label: '一级2' },
              { id: 21, parentId: 2, label: '二级2-1' },
              { id: 22, parentId: 2, label: '二级2-2' },
            ])
          }, 300)
        })
      },
    }
  }
</script>

:::

# 集成树搜索

  • 设置show-search后即启用默认的树搜索功能，懒加载时同样设置。
  • 自定义搜索框文本searchLabel和文本宽度labelWidth

:::demo

<template>
  <ds-tree
    v-model="value"
    :data="data"
    dataType="parallel"
    searchLabel="组织:"
    labelWidth="60"
    show-search
   ></ds-tree>
</template>
<script>
  export default {
    data() {
      return {
        value: '',
        data: [
          { id: 1, parentId: 0, label: '一级1' },
          { id: 11, parentId: 1, label: '二级1-1' },
          { id: 111, parentId: 11, label: '二级1-1-1' },
          { id: 112, parentId: 11, label: '二级1-1-2' },
          { id: 2, parentId: 0, label: '一级2' },
          { id: 21, parentId: 2, label: '二级2-1' },
          { id: 22, parentId: 2, label: '二级2-2' },
        ],
      }
    },
  }
</script>

:::

还可以传入queryFilter自定义树搜索行为。

queryFilter中会抛出当前搜索值和elTree的数据中心store。

由于无法预测queryFilter中是否有异步操作，故统一return一个promise。

请自行处理成树结构数据再resolve给组件。

:::demo

<template>
  <ds-tree v-model="value" show-search :data="data" :queryFilter="queryFilter" dataType="parallel"></ds-tree>
</template>
<script>
  export default {
    data() {
      return {
        value: '',
        data: [
          { id: 1, parentId: 0, label: '一级1' },
          { id: 11, parentId: 1, label: '二级1-1' },
          { id: 111, parentId: 11, label: '二级1-1-1' },
          { id: 112, parentId: 11, label: '二级1-1-2' },
          { id: 2, parentId: 0, label: '一级2' },
          { id: 21, parentId: 2, label: '二级2-1' },
          { id: 22, parentId: 2, label: '二级2-2' },
        ],
      }
    },
    methods: {
      queryFilter(query, store) {
        return new Promise((resolve) => {
          setTimeout(() => {
            // do something...
            // 请自行处理成树结构再返回
            // 这里为了演示效果直接写成死数据
            resolve([
              { 
                id: 11,
                parentId: 1,
                label: '二级1-1',
                children: [
                  { id: 111, parentId: 11, label: '二级1-1-1' },
                  { id: 112, parentId: 11, label: '二级1-1-2' },
                ]
              },
            ])
          }, 0)
        })
      },
    }
  }
</script>

:::

# Tree props

参数 说明 类型 可选值 默认值
value/v-model 搜索条件 String - -
placeholder 搜索输入框的placeholder String - -
data 树渲染数据 Array - -
dataType 数据结构 String 'tree'/'parallel' 'tree'
showSearch 是否使用树搜索功能 Boolean true/false true
searchLabel 搜索文本 String - -
labelWidth 搜索文本宽度 String - 40
dataProps id/parentId字段映射 Object - { id: 'id', parentId: 'parentId'}
queryFilter 自定义搜索行为 Function - -
lazy 是否懒加载 Boolean true/false false
...elTree.$props 继承el-tree`所有属性 - - -
