# Table 表格

表格配置化

# 基础用法

  • 列配置对象会继承el-table-column所有属性
  • 支持自定义动态渲染
  • 拦截formatter抛出el-table-column的slotScope :::demo
<template>
  <ds-table v-bind="tableProps"></ds-table>
</template>
<script>
  export default {
   data() {
      return {
        tableProps: {
          data: [
            { name: 'zhonggz1', age: 11, height: '171', phone: '111111111', date: '2021-1-1', address: 'test1' },
            { name: 'zhonggz2', age: 12, height: '172', phone: '111111112', date: '2021-1-1', address: 'test2' },
            { name: 'zhonggz3', age: 13, height: '173', phone: '111111113', date: '2021-1-1', address: 'test3' },
            { name: 'zhonggz4', age: 14, height: '174', phone: '111111114', date: '2021-1-1', address: 'test4' },
            { name: 'zhonggz5', age: 15, height: '175', phone: '111111115', date: '2021-1-1', address: 'test5' }
          ],
          columns: [
            { label: '姓名', prop: 'name' },
            { label: '年龄', prop: 'age', formatter: ({row}) => `${row.age}岁` },
            { label: '身高', prop: 'height', sortable: true },
            { label: '手机', prop: 'phone', render: (h, vm) => <el-button onClick={() => console.log(`${vm.displayValue}`)}>{vm.displayValue}</el-button> },
            { label: '日期', prop: 'date' },
            { label: '地址', prop: 'address' },
          ],
        }
      }
    },
  }
</script>

# 表格头部操作栏

同ds-toolbar的用法

:::demo

<template>
  <ds-table v-bind="tableProps"></ds-table>
</template>
<script>
  export default {
    data() {
      return {
        tableProps: {
          data: [
            { name: 'zhonggz1', age: 11, height: '171', phone: '111111111', date: '2021-1-1', address: 'test1' },
            { name: 'zhonggz2', age: 12, height: '172', phone: '111111112', date: '2021-1-1', address: 'test2' },
            { name: 'zhonggz3', age: 13, height: '173', phone: '111111113', date: '2021-1-1', address: 'test3' },
            { name: 'zhonggz4', age: 14, height: '174', phone: '111111114', date: '2021-1-1', address: 'test4' },
            { name: 'zhonggz5', age: 15, height: '175', phone: '111111115', date: '2021-1-1', address: 'test5' }
          ],
          columns: [
            { type: 'selection' },
            { type: 'index' },
            { label: '姓名', prop: 'name' },
            { label: '年龄', prop: 'age' },
            { label: '身高', prop: 'height' },
            { label: '手机', prop: 'phone' },
            { label: '日期', prop: 'date' },
            { label: '地址', prop: 'address' },
          ],
          toolbar: {
            align: 'left',
            tools: [
              {
                text: '新增',
                props: {
                  type: 'primary'
                },
                on: {
                  click: () => {
                    let data = { id: 6, name: 'zhonggz6', age: 16, height: '176', phone: '111111116', date: '2021-1-1', address: 'test6' }
                    this.tableProps.data.push(data)
                  }
                }
              },
              {
                text: '删除',
                on: {
                  click: vm => {
                    vm.getSelection.forEach(row => {
                      let idx = this.tableProps.data.indexOf(row)
                      idx > -1 && this.tableProps.data.splice(idx, 1)
                    })
                  }
                }
              },
              {
                text: '左对齐',
                on: {
                  click: () => {
                    this.tableProps.toolbar.align = 'left';
                  }
                }
              },
              {
                text: '右对齐',
                on: {
                  click: () => {
                    this.tableProps.toolbar.align = 'right';
                  }
                }
              }
            ]
          }
        }
      }
    },
  }
</script>

# 集成分页

  • 通过showPagination启用分页功能
  • 从paginationProps传入所有el-pagination属性
  • 当pageIndex或pageSize 发生改变时, 统一抛出 page-change事件并返回分页数据

:::demo

<template>
  <ds-table v-bind="tableProps" showPagination @page-change='handlePageChange'></ds-table>
</template>
<script>
  export default {
    data() {
      return {
        tableProps: {
          data: [
            { name: 'zhonggz1', age: 11, height: '171', phone: '111111111', date: '2021-1-1', address: 'test1' },
            { name: 'zhonggz2', age: 12, height: '172', phone: '111111112', date: '2021-1-1', address: 'test2' },
            { name: 'zhonggz3', age: 13, height: '173', phone: '111111113', date: '2021-1-1', address: 'test3' },
            { name: 'zhonggz4', age: 14, height: '174', phone: '111111114', date: '2021-1-1', address: 'test4' },
            { name: 'zhonggz5', age: 15, height: '175', phone: '111111115', date: '2021-1-1', address: 'test5' },
          ],
          columns: [
            { label: '姓名', prop: 'name' },
            { label: '年龄', prop: 'age' },
            { label: '身高', prop: 'height' },
            { label: '手机', prop: 'phone' },
            { label: '日期', prop: 'date' },
            { label: '地址', prop: 'address' },
          ],
          paginationProps: {
            // pageIndex: 1, // => current-page
            // pageSize: 10,
            total: 5, // 接口返回总行数
            // align: 'right', // 默认右对齐
            // layout: 'total, sizes, prev, pager, next, jumper', // 默认分页布局
            // pagerCount: 5, // 页码按钮的数量，当总页数超过该值时会折叠
            pageSizes: [5, 10], // 每页显示个数选择器的选项设置
          }
        }
      }
    },
    methods: {
      handlePageChange(info) {
        this.$message(JSON.stringify(info));
      }
    }
  }
</script>

# 行内编辑

在columns配置对象中增加editor编辑配置将激活行内编辑功能。

内置type: text/input/select/date/dateRange/switch/number控件。

注意:

data中的每一行数据row都会在组件内部维护一个_editing属性，用于控制行内编辑状态。

新增行默认为编辑状态，其他行默认为只读状态。

另外，该_editing状态也可以手动在外部传入。见下表新增按钮事例代码。

:::demo

<template>
  <ds-table v-bind="tableProps"></ds-table>
</template>
<script>
  export default {
    data() {
      return {
        tableProps: {
          data: [],
          columns: [
            { type: 'index' },
            { type: 'selection' },
            { label: '姓名', prop: 'name' },
            { label: '年龄', prop: 'age' },
            { label: '身高', prop: 'height' },
            {
              label: '手机',
              prop: 'phone',
              editor: {
                type: 'select',
                props: {
                  placeholder: '请选择手机'
                },
                options: [
                  { label: '11111111', value: '11111111' },
                  { label: '11111112', value: '11111112' },
                  { label: '11111113', value: '11111113' },
                  { label: '11111114', value: '11111114' },
                  { label: '11111115', value: '11111115' },
                ],
                on: {
                  change: this.selectChange
                },
              },
            },
            {
              label: '日期',
              prop: 'date',
              editor: {
                type: 'date',
                props: {
                  placeholder: '请选择日期'
                },
                on: {
                  change: this.dateChange
                },
              }
            },
            {
              label: '地址',
              prop: 'address',
              editor: {
                type: 'input',
                props: {
                  placeholder: '请输入地址'
                },
              }
            },
          ],
          toolbar: {
            align: 'left',
            tools: [
              {
                text: '新增',
                props: {
                  type: 'primary'
                },
                on: {
                  click: () => {
                    let data = { id: 6, name: 'tom6', age: 16, height: '176', phone: '11111116', date: '', address: '', _editing: true}
                    this.tableProps.data.push(data)
                  }
                }
              },
            ]
          },
          operator: [
            {
              text: '查看详情',
              props: {
                type: 'text',
              },
              on: {
                click: ({row}) => {
                  this.$message.success(JSON.stringify(row));
                }
              }
            },
            cell => ({
              text: cell.row._editing ? '保存' : '编辑',

                type: 'text',
              },
              on: {
                click: cell => {
                  cell.row._editing = !cell.row._editing;
                }
              }
            }),
          ],
          operatorProps: {
            width: 100
          },
        }
      }
    },
    created () {
      this.getData()
    },
    methods: {
      getData() {
        setTimeout(() => {
          this.tableProps.data = [
            { name: 'zhonggz1', age: 11, height: '171', phone: '11111111', date: '2021-1-1', address: 'test1' },
            { name: 'zhonggz2', age: 12, height: '172', phone: '11111112', date: '2021-1-1', address: 'test2' },
            { name: 'zhonggz3', age: 13, height: '173', phone: '11111113', date: '2021-1-1', address: 'test3' },
            { name: 'zhonggz4', age: 14, height: '174', phone: '11111114', date: '2021-1-1', address: 'test4' },
            { name: 'zhonggz5', age: 15, height: '175', phone: '', date: '', address: '' }

# 行内校验

提供了跟el-form一样校验的功能，需要在editor的 rules 属性传入约定的校验规则。

ds-table内部实现了rowValidate行校验方法，它接收两个参数：需要校验的行数index和回调(valid, res)=>{}。

:::demo

<template>
  <ds-table v-bind="tableProps"></ds-table>
</template>
<script>
  export default {
    data() {
      return {
        tableProps: {
          data: [],
          columns: [
            { type: 'index' },
            { type: 'selection' },
            { label: '姓名', prop: 'name' },
            { label: '年龄', prop: 'age' },
            { label: '身高', prop: 'height' },
            {
              label: '手机',
              prop: 'phone',
              editor: {
                type: 'select',
                options: [
                  { label: '11111111', value: '11111111' },
                  { label: '11111112', value: '11111112' },
                  { label: '11111113', value: '11111113' },
                  { label: '11111114', value: '11111114' },
                  { label: '11111115', value: '11111115' },
                ],
                rules: [{
                  required: true,
                  trigger: ['blur', 'change']
                }],
              },
            },
            {
              label: '日期',
              prop: 'date',
              editor: {
                type: 'date',
                props: {
                  placeholder: '请选择日期'
                },
                rules: [{
                  required: true,
                  trigger: ['blur', 'change']
                }],
              }
            },
            {
              label: '地址',
              prop: 'address',
              editor: {
                type: 'input',
                props: {
                  placeholder: '请输入地址'
                },
                rules: {
                  required: true,
                }
              }
            },
          ],
          operator: [
            cell => ({
              text: cell.row._editing ? '保存' : '编辑',
              props: {
                type: 'text',
              },
              on: {
                click: cell => {
                  cell._self.rowValidate(cell.$index, (valid, res) => {
                    if (valid) {
                      cell.row._editing = !cell.row._editing;
                    }
                  })
                }
              }
            }),
          ],
        }
      }
    },
    created () {
      this.getData()
    },
    methods: {
      getData() {
        setTimeout(() => {
          this.tableProps.data = [
            { name: 'zhonggz1', age: 11, height: '171', phone: '11111111', date: '2021-1-1', address: 'test1' },
            { name: 'zhonggz2', age: 12, height: '172', phone: '11111112', date: '2021-1-1', address: 'test2' },
            { name: 'zhonggz3', age: 13, height: '173', phone: '11111113', date: '2021-1-1', address: 'test3' },
            { name: 'zhonggz4', age: 14, height: '174', phone: '11111114', date: '2021-1-1', address: 'test4' },
            { name: 'zhonggz5', age: 15, height: '175', phone: '', date: '', address: '' }
          ]
        }, 500)
      },
    }
  }
</script>

# Table props

参数 说明 类型 可选值 默认值
data 表格数据 Array - []
columns 列配置对象（colunm-item）的集合，具体配置见下表 Array - []
operator 操作按钮配置对象的集合，同 ds-toolbar Array - -
operatorProps 继承el-table-colunm的所有属性 Object - -
toolbar 表格头部操作栏，同 ds-toolbar Object - -
showPagination 是否分页 Boolean - false
paginationProps 继承el-pagination的所有属性 Object - -
renderFooter 表格底部自定义渲染 Object/Function - -
autoFix 全表格自适应列宽 Boolean true/false False
...elTable.$props 继承el-table所有属性 - - -

# Colunm-Item Props

参数 说明 类型 可选值 默认值
type el-table特殊列 String index/selection -
label 表头 String - -
prop 绑定字段 String - -
render 自定义jsx动态渲染，并抛出当前实例 Function (h, vm) => {} -
formatter 格式化数据, 会抛出el-table-colunm的slotScope Function slotScope => {} -
editor 编辑项配置对象，具体见下表 Object - -
hidden 列隐藏 Boolean true/false -
autoFix 自适应列宽 Boolean true/false False
...elTableColunm.$props 继承el-table-colunm 所有属性 - - -

# Editor Props

参数 说明 类型 可选值 默认值
type 内置编辑控件 String text/input/select/date/dateRange/switch/number -
props 继承当前type指定的控件的所有属性 Object - -
on 监听当前type指定的控件抛出的所有事件 Object - -
options 当type为select时要传入此选项 Array - -
rules 同el-form 的校验规则 Array/Object - -

# Table Methods

方法名 说明 参数 参数类型 回调参数
validate 表格校验 cb Function cb(valid,validMsg) => {}
rowValidate 行校验 Index,cb String,Object cb(valid,validMsg) => {}
getSelection 获取选择项 - - -

事件 说明 参数
