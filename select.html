# Select 选择器

增强el-select，主要是不想再去写v-forel-option了。

# 基础用法

:::demo

<template>
  <ds-select v-model="selected" v-bind="selectProps" @change='handlChange'></ds-select>
</template>
<script>
  export default {
    data() {
      return {
        selected: '',
        selectProps: {
          options: [
            { label: '张三', value: 'zhangsan' },
            { label: '李四', value: 'lisi' },
            { label: '王五', value: 'wangwu' },
            { label: '赵六', value: 'zhaoliu' }
          ]
        }
      }
    },
    methods: {
      handlChange(val) {
        this.$message.success(val)
        this.selectProps.options.push({ label: '李七', value: 'liqi' })
      },
    }
  }
</script>

:::

# 远程搜索

远程搜索后组件内部会自动更新下拉选项。

又因为异步，querySearch必须return一个Promise，并且把返回数据resolve给组件内部使用。

:::demo

<template>
  <ds-select v-model="selected" v-bind="selectProps"></ds-select>
</template>
<script>
  export default {
    data() {
      return {
        selected: '',
        selectProps: {
          options: [
            { label: '张三', value: 'zhangsan' },
            { label: '李四', value: 'lisi' },
            { label: '王五', value: 'wangwu' },
            { label: '赵六', value: 'zhaoliu' }
          ],
          querySearch: this.querySearch,
        }
      }
    },
    methods: {
      querySearch(query) { 
        return new Promise((resolve, reject) => {
          setTimeout(() => {
            const list = this.selectProps.options.filter(item => item.label.indexOf(query) > -1);
            resolve(list)
          }, 300);
        })
      },
    }
  }
</script>

:::

# 只读模式

编辑表单时可能用得着

:::demo

<template>
  <ds-select v-model="selected" :options="options" :readonly='readonly' autoShow></ds-select>
  <ds-toolbar :tools='tools' style="marginTop: 8px"></ds-toolbar>
</template>
<script>
  export default {
    data() {
      return {
        selected: '',
        readonly: false,
        options: [
          { label: '张三', value: 'zhangsan' },
          { label: '李四', value: 'lisi' },
          { label: '王五', value: 'wangwu' },
          { label: '赵六', value: 'zhaoliu' }
        ],
      }
    },
    computed: {
      tools() {
        return [
          {
            text: this.readonly ? '编辑' : '只读',
            on: {
              click: () => this.readonly = !this.readonly
            }
          }
        ]
      }
    },
  }
</script>

:::

# tooltip

超长默认显示tooltip :::demo

<template>
  <ds-select v-model="selected" :options="options" :tooltip='tooltip' autoShow></ds-select>
  <ds-toolbar :tools='tools' style="marginTop: 8px"></ds-toolbar>
</template>
<script>
  export default {
    data() {
      return {
        selected: '',
        tooltip: true,
        options: [
          { label: '张三张三张三张三张三张三张三张三张三张三张三张三张三张三', value: 'zhangsan' },
          { label: '李四', value: 'lisi' },
          { label: '王五', value: 'wangwu' },
          { label: '赵六', value: 'zhaoliu' }
        ],
      }
    },
    computed: {
      tools() {
        return [
          {
            text: this.tooltip ? '隐藏tooltip' : '显示tooltip',
            on: {
              click: () => this.tooltip = !this.tooltip
            }
          }
        ]
      }
    },
  }
</script>

:::

# 虚拟滚动

当你觉得下拉选项的数据量太大导致影响性能时使用 :::demo

<template>
  <ds-select v-model="selected" :options="options" virtual></ds-select>
</template>
<script>
  const options = Array.from({length: 100000}, (data, index) => ({
    label: `zhonggz${index}`,
    value: index
  }))

  export default {
    data() {
      return {
        selected: '',
        options,
      }
    },
  }
</script>

:::

# 绑定对象或对象数组

如果ds-select的绑定值为对象类型，请务必指定 value-key 作为它的唯一标识。

绑定对象数组时还需要同时设置multiple。

:::demo

<template>
  <ds-select v-model="selected1" v-bind="selectProps1" :value-key="valueKey" :props="props"></ds-select>
  <div style="margin-top: 10px">v-model：{{selected1}}</div>
  <br/>
  <br/>
  <ds-select v-model="selected2" v-bind="selectProps2" :value-key="valueKey" :props="props"></ds-select>
  <div style="margin-top: 10px">v-model：{{selected2}}</div>
</template>
<script>
  export default {
    data() {
      return {
        valueKey: 'code',
        props: {
          label: 'name',
          value: 'code',
        },
        selected1: { name: '张三', code: 'zhangsan' },
        selected2: [
          { name: '张三', code: 'zhangsan' },
          { name: '李四', code: 'lisi' }
        ],
        selectProps1: {
          options: [
            { name: '张三', code: 'zhangsan' },
            { name: '李四', code: 'lisi' },
            { name: '王五', code: 'wangwu' },
            { name: '赵六', code: 'zhaoliu' }
          ],
        },
        selectProps2: {
          options: [
            { name: '张三', code: 'zhangsan' },
            { name: '李四', code: 'lisi' },
            { name: '王五', code: 'wangwu' },
            { name: '赵六', code: 'zhaoliu' }
          ],
          multiple: true,
        },
      }
    },
  }
</script>

:::

# Select Props

参数 说明 类型 可选值 默认值
v-model 绑定值 String - ''
options 下拉选项集合 Array - []
optionKey 自定义option的key,默认取value值值 Function - -
autoShow 自动显示第一项 Boolean true/false false
props el-option的label/value映射 Object {label: '', value: ''} -
querySearch 远程搜索 function (query) => Promise.resolve(list) -
tooltip 是否显示超长tooltip Boolean true/false -
optionKey 设置option的key值 Function (option, index) => {} -
virtual 开启虚拟滚动 Boolean true/false false
valueKey 绑定对象类型时必须指定valueKey作为唯一标识 String - -
...elSelect.$props 继承el-select所有属性 - - -
