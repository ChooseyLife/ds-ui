# Form 表单

表单配置化

# 基础用法

  • 内置输入框，选择器，日期，日期范围，日期时间，开关，金额输入框， 级联，单选，多选表单控件，不需要再关心HTML代码。
  • 自动完成双向绑定。
  • 同时支持自定义渲染。
  • 表单项配置支持函数写法。
  • 增加表单控件尾部插槽append。

::: demo

<template>
  <ds-container>
    <ds-title>基本信息</ds-title>
    <ds-form v-bind="formProps" :itemConfig='itemConfig' ref='form'></ds-form>
    <div>{{formProps.model}}</div>
  </ds-container>
</template>
<script>
const cascaderOpt = [{
  value: 'ziyuan',
  label: '资源',
  children: [{
    value: 'axure',
    label: 'Axure Components'
  }, {
    value: 'sketch',
    label: 'Sketch Templates'
  }, {
    value: 'jiaohu',
    label: '组件交互文档'
  }]
}]

  export default {
    data() {
      return {
        hiddenInput: false,
        selectOpts: [
          { value: '选项1', label: '黄金糕'},
          { value: '选项2', label: '双皮奶'}
        ],
        formProps: {
          model: {
            a: '',
            b: '',
            c: '',
            d: '',
            e: false,
            f: undefined,
            g: '',
            h: '',
            i: [],
            j: '',
            foo: '',
            append: '',
            appendRender: '',
          },
        }
      }
    },
    computed: {
      itemConfig() {
        return [
          {
            label: '输入框',
            prop: 'a',
            type: 'input',
            props: {
              number: true,
            },
            hidden: this.hiddenInput,
          },
          {
            label: '选择器',
            prop: 'b',
            type: 'select',
            options: this.selectOpts,
          },
          {
            label: '日期',
            prop: 'c',
            type: 'date',
            props: {
              pickerOptions: {
                disabledDate(time) {
                  return time.getTime() > Date.now();
                }
              },
            },
          },
          {
            label: '日期时间',
            prop: 'd',
            type: 'dateTime',
          },
          {
            label: '日期范围',
            prop: 'd',
            type: 'dateRange',
            props: {
              valueFormat: 'yyyy-MM-dd'
            },
          },
          {
            label: '开关',
            prop: 'e',
            type: 'switch',
          },
          {
            label: '金额输入',
            prop: 'f',
            type: 'number',
            props: {
              amount: true,
              clearable: true,
            },
          },
          {
            label: '级联',
            prop: 'g',
            type: 'cascader',
            options: cascaderOpt,
          },
          {
            label: '单选',
            prop: 'h',
            type: 'radio',
            options: [
              { value: 'm', label: '男'},
              { value: 'f', label: '女'}
            ],
          },
          {
            label: '多选',
            prop: 'i',
            type: 'checkbox',
            options: [
              { value: 'm', label: '男'},
              { value: 'f', label: '女'}
            ],
          },
          {
            label: '自定义',
            type: 'custom',
            render: () => {
              return (
                <el-button onClick={this.hidden}>{`${this.hiddenInput ? '显示' : '隐藏' }输入框`}</el-button>
              )
            },
          },
          {
            label: '自定义',
            type: 'custom',
            render: () => {
              return (
                <el-button onClick={this.addSelectOpts}>增加选择器项选</el-button>
              )
            },
          },
          vm => {
            return {
              label: '函数写法',
              prop: 'foo',
              type: 'input',
            }
          },
          {
            label: 'append',
            type: 'input',
            prop: 'append',
            append: 'kg'
          },
          {
            label: '自定义append',
            type: 'input',
            prop: 'appendRender',
            append: () => <i class='el-icon-warning'></i>,
          },
        ]
      }
    },
    methods: {
      hidden() {
        this.hiddenInput = !this.hiddenInput;
      },
      addSelectOpts() {
        this.selectOpts.push({ value: '选项3', label: '绿豆沙'});
      }
    }
  }
</script>

# 表单栅格系统

基于el-col/el-row封装, 整个表单将划分为24列。

可以整体配置, 默认分为3列。

:::demo

<template>
  <ds-form v-bind="formProps" ref='form'></ds-form>
  <ds-toolbar :tools='tools'></ds-toolbar>
</template>
<script>
  export default {
    data() {
      return {
        formProps: {
          col: 3,
          model: {
            a: '',
            b: '',
            c: '',
            d: '',
            e: '',
            f: '',
            g: '',
          },
          itemConfig: [
            { label: 'A', prop: 'a', type: 'input' },
            { label: 'B', prop: 'b', type: 'input' },
            { label: 'C', prop: 'c', type: 'input' },
            { label: 'D', prop: 'd', type: 'input' },
            { label: 'E', prop: 'e', type: 'input' },
            { label: 'F', prop: 'f', type: 'input' },
            { label: 'G', prop: 'g', type: 'input' },
          ]
        },
        tools: [
          { text: '1列', on: { click: vm => { vm.formProps.col = 1 }}},
          { text: '2列', on: { click: vm => { vm.formProps.col = 2 }}},
          { text: '3列', on: { click: vm => { vm.formProps.col = 3 }}},
        ]
      }
    }
  }
</script>

:::

# 表单校验

继承el-form校验行为

:::demo

<template>
  <ds-form v-bind="formProps" ref='form'></ds-form>
  <el-button @click='valide'>校验</el-button>
</template>
<script>
  export default {
    data() {
      return {
        formProps: {
          model: {
            a: '',
            b: '',
          },
          itemConfig: [
            { label: 'A', prop: 'a', type: 'input' },
            { label: 'B', prop: 'b', type: 'select', options: [
              { value: '选项1', label: '黄金糕'},
              { value: '选项2', label: '双皮奶'}
            ] },
          ],
          rules: {
            a: [{ required: true, message: '请输入', trigger: 'change' }],
            b: [{ required: true, message: '请选择', trigger: 'change' }],
          },
        },
      }
    },
    methods: {
      valide() {
        this.$refs.form.validate((valid) => {
          if (valid) {
            this.$message.success(JSON.stringify(valid));
          } else {
            this.$message.error(JSON.stringify(valid));
          }
        })
      }
    }
  }
</script>

# 含有禁用表单时的校验行为

  • 在表单编辑时可能会出现部分字段数据丢失，如果该字段编辑时禁止修改，那么表单将永远无法通过校验。
  • v0.4.18: 可以通过配置dontCheckDisabled属性跳过禁用表单的校验。

:::demo

<template>
  <ds-form v-bind="formProps" dontCheckDisabled ref='form'></ds-form>
  <div style='marginBottom: 16px;'>{{formProps.model}}</div>
  <ds-toolbar :tools='tools'></ds-toolbar>
</template>
<script>
  export default {
    data() {
      return {
        tools: [
          {
            text: '跳过禁用表单的校验',
            props: {
              type: 'primary',
            },
            on: {
              click: this.validate,
            }
          },
        ],
        formProps: {
          model: {
            a: '',
            b: '',
          },
          itemConfig: [
            { label: 'A', prop: 'a', type: 'input', props: { disabled: true } },
            { label: 'B', prop: 'b', type: 'select', options: [
              { value: 'hjg', label: '黄金糕'},
              { value: 'spn', label: '双皮奶'}
            ] },
          ],
          rules: {
            a: [{ required: true, message: '请输入', trigger: 'change' }],
            b: [{ required: true, message: '请选择', trigger: 'change' }],
          },
        },
      }
    },
    created() {
      this.initEditData();
    },
    methods: {
      initEditData() {
        const data = {
          a: '',
          b: 'hjg'
        }
        Promise.resolve(data).then(data => {
          Object.assign(this.formProps.model, data);
        })
      },
      validate() {
        this.$refs.form.validate((valid) => {
          if (valid) {
            this.$message.success(JSON.stringify(valid));
          }
        })
      }
    }
  }
</script>

:::

# search模式展开与收起状态

  • search模式下，表单控件超过一行会显示展开/收起，点击文本按钮切换状态， 默认展开。
  • 可通过collapse属性修改默认展开/收起状态。 ::: demo
<template>
  <ds-form v-bind="formProps" search :collapse="isCollapse"></ds-form>
  <el-button @click='changeCollapse'>changeCollapse</el-button>
</template>
<script>
  export default {
    data() {
      return {
        isCollapse: true,
        formProps: {
          model: {
            a: '',
            b: '',
            c: '',
            d: '',
            e: '',
            f: '',
          },
          itemConfig: [
            { label: 'A', prop: 'a', type: 'input' },
            { label: 'B', prop: 'b', type: 'input' },
            { label: 'C', prop: 'c', type: 'input' },
            { label: 'D', prop: 'd', type: 'input' },
            { label: 'E', prop: 'e', type: 'input' },
            { label: 'F', prop: 'f', type: 'select', options: [
              { value: '选项1', label: '黄金糕'},
              { value: '选项2', label: '双皮奶'}
            ] },
          ],
        },
      }
    },
    methods: {
      changeCollapse() {
        this.isCollapse = !this.isCollapse;
      }
    }
  }
</script>

:::

# Form props

参数 说明 类型 可选值 默认值
col 表单布局列数 Number 能被24整除的数 3
model 表单绑定值，同el-form Object - -
itemConfig 表单项配置对象（form-item）的集合或者通过函数返回, 具体配置见下表 Array/Function - -
rules 校验规则，同el-form Object - -
labelWidth form-item的label宽度 Number - 90
search 搜索模式 Boolean true/false false
collapse 控制搜索模式时表单的展开/收起状态，默认展开 Boolean true/false false
readonly 只读模式 Boolean true/false False
dontCheckDisabled 跳过禁用表单的校验 Boolean true/false False
...elForm.$props 继承el-form的所有属性 - - -

# Form Events

事件 说明 参数
search 搜索时触发 model
reset 重置时触发 -
collapse 展开或收起时触发 true/false

# Form Methods

方法名 说明 参数 参数类型 返回
getModel 获取表单绑定的值 - - model
setModel 设置表单绑定的值 model Object -
setFieldReadonly 将表单项设置为只读 prop String -
removeFieldReadonly 移除表单项只读设置 prop String -

# Form-Item porps

参数 说明 类型 可选值 默认值
type 内置表单项 String text/input/select/date/dateRange/switch
/number/cascader/radio/checkbox		 -
label 同el-form-item String - -
prop 同el-form-item String - -
span 表单项所占列数 Number [1-24] -
props 继承当前type指定的控件的所有属性 Object - -
on 监听当前type指定的控件抛出的所有事件 Object - -
options 当type为select/cascader/radio
/checkbox时要传入选项		 Array - -
render 自定义jsx动态渲染 Function h => () -
append 表单控件尾部插槽 Function/String h => ()/'' -
hidden 控制表单项的显示隐藏 Boolean true/false -
readonly 控制表单项的只读/编辑 Boolean true/false -
